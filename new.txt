const express = require("express");
const path = require("path");
const cors = require("cors");
const mongoose = require("mongoose");
const session = require("express-session");
const rateLimit = require("express-rate-limit");
const morgan = require("morgan");
const dotenv = require("dotenv");
const fs = require("fs");
const Joi = require("joi");
const MongoStore = require("connect-mongo");

dotenv.config();

const app = express();

// âœ… Step 1: Validate Environment Variables
const envSchema = Joi.object({
  MONGO_URI: Joi.string().required(),
  SESSION_SECRET: Joi.string().min(10).required(),
  NODE_ENV: Joi.string().valid("development", "production").default("development"),
  PORT: Joi.number().default(8080),
}).unknown();

const { error, value: envVars } = envSchema.validate(process.env);
if (error) {
  console.error("âŒ Invalid .env configuration:", error.message);
  process.exit(1);
}

// âœ… Step 2: Connect MongoDB
mongoose.connect(envVars.MONGO_URI)
  .then(() => console.log("âœ… MongoDB connected"))
  .catch(err => {
    console.error("âŒ MongoDB connection error:", err);
    process.exit(1);
  });

// âœ… Step 3: Logging
app.use(morgan("dev"));

// âœ… Step 4: Rate Limiting on Login
const loginLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 10,
  message: "âš  Too many login attempts. Please try again later.",
  standardHeaders: true,
  legacyHeaders: false
});
app.use("/admin-portal-1024/login-secret", loginLimiter);

// âœ… Step 5: Secure Session Handling
app.use(session({
  secret: envVars.SESSION_SECRET,
  resave: false,
  saveUninitialized: false,
  rolling: true, // â± Refresh session on every request
  store: MongoStore.create({
    mongoUrl: envVars.MONGO_URI,
    collectionName: "adminSessions",
    ttl: 7 * 24 * 60 * 60 // 7 days
  }),
  cookie: {
    httpOnly: true,
    secure: envVars.NODE_ENV === "production",
    sameSite: "strict",
    maxAge: 7 * 24 * 60 * 60 * 1000 // 7 days
  }
}));

// âœ… Step 6: Basic Middleware
app.use(cors());
app.use(express.urlencoded({ extended: true }));
app.use(express.json());

// âœ… Step 7: Views and Static
app.set("view engine", "ejs");
app.set("views", path.join(__dirname, "views"));
app.use(express.static(path.join(__dirname, "public")));

// âœ… Step 8: Routes
const paymentRoutes = require("./routes/payments.routes");
const adminRoutes = require("./routes/admin.routes");

app.use("/api", paymentRoutes);
app.use("/admin-portal-1024", adminRoutes);

// âœ… Step 9: Public Pages
app.get("/", (req, res) => res.render("home"));
app.get("/product", (req, res) => res.render("product"));
app.get("/form", (req, res) => res.render("form"));
app.get("/about", (req, res) => res.render("about"));
app.get("/reviews", (req, res) => res.render("reviews"));
app.get("/T&C", (req, res) => res.render("T&C"));

// âœ… Step 10: 404 Handler
app.use((req, res) => {
  res.status(404).render("404", { title: "Page Not Found" });
});

// âœ… Step 11: Start Server
const PORT = envVars.PORT;
app.listen(PORT, () => {
  console.log(`ðŸš€ Server running at http://localhost:${PORT}`);
});
